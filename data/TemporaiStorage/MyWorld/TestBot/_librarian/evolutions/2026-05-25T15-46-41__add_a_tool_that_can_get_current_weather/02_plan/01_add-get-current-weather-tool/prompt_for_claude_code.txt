   @tool([AGENT_NAME])
   def get_current_weather(self: Mind, location: str = "") -> str:

       Use when: the user asks about the weather, temperature, conditions,
       humidity, or wind for a specific city, region, or place.
       Do NOT use for historical weather or multi-day forecasts.

       location: City or place name (e.g. "London", "New York", "Tokyo").
       """
       if not location:
           return "Error: 'location' parameter is required. Provide a city or place name."

       encoded = urllib.parse.quote(location)
       url = f"https://wttr.in/{encoded}?format=j1"

       try:
           req = urllib.request.Request(url, headers={"User-Agent": "TestBot/1.0"})
           with urllib.request.urlopen(req, timeout=10) as resp:
               data = json.loads(resp.read().decode())
       except urllib.error.URLError as exc:
           return json.dumps({"error": f"Could not reach weather service: {exc.reason}"})
       except Exception as exc:
           return json.dumps({"error": str(exc)})

       current = data.get("current_condition", [{}])[0]
       desc = current.get("weatherDesc", [{}])[0].get("value", "Unknown")
       nearest = data.get("nearest_area", [{}])[0]
       area = nearest.get("areaName", [{}])[0].get("value", location)
       country = nearest.get("country", [{}])[0].get("value", "")

       return json.dumps({
           "location": f"{area}, {country}" if country else area,
           "condition": desc,
           "temperature_c": current.get("temp_C", "?"),
           "temperature_f": current.get("temp_F", "?"),
           "feels_like_c": current.get("FeelsLikeC", "?"),
           "feels_like_f": current.get("FeelsLikeF", "?"),

           "visibility_km": current.get("visibility", "?"),
           "uv_index": current.get("uvIndex", "?"),
       })
